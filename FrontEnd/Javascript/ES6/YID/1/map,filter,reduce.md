# reduce
```js
const reduce = (f, acc, iter) => {
    for(const a of iter){
        acc = f(acc, a);
    }
    return acc;

}

const add = (a, b) => a + b;
console.log(reduce(add, 0, [1,2,3,4,5]));
// 15

//reduce는 재귀함수이다.
console.log(add(add(add(add(add(0, 1), 2), 3),4),5));
//15
```
